    function drawNatalChartWheel(sunLong, moonLong, ascLong, planets, cusps) {
      const svg = document.getElementById("natalChartSvg");
      svg.innerHTML = ''; // Temizle

      const cx = 250, cy = 250;
      const rOuter = 220, rMid = 190, rInner = 100;

      // 1. Dв”Ђв–’в”јЕџ halkalarв”Ђв–’ в”њДџiz
      svg.innerHTML += `<circle cx="${cx}" cy="${cy}" r="${rOuter}" stroke="var(--gold)" stroke-width="2" fill="none" />`;
      svg.innerHTML += `<circle cx="${cx}" cy="${cy}" r="${rMid}" stroke="#1f2937" stroke-width="1.5" fill="none" />`;
      svg.innerHTML += `<circle cx="${cx}" cy="${cy}" r="${rInner}" stroke="#1f2937" stroke-width="1.5" fill="none" />`;

      // 2. 12 Evi ve Burв”њДџ Dilimlerini в”њГ§iz (Whole Sign tabanlв”Ђв–’, ASC sol ufuk в”њДџizgisinde = 180 derecede olacak в”јЕџekilde)
      // ASC'nin zodyaktaki derecesi
      const ascSignIdx = Math.floor(ascLong / 30);
      const ascSignStartLong = ascSignIdx * 30;

      // Her evin в”њДџizgisini в”њДџiz
      for (let i = 0; i < 12; i++) {
        const cuspLong = (ascSignIdx + i) * 30;
        // ASC sol ufuktadв”Ђв–’r (180в”¬в–‘), o yв”њв•ќzden relatif aв”њДџв”Ђв–’ hesaplanв”Ђв–’r
        const angleDeg = 180 - (cuspLong - ascLong);
        const angleRad = AstroCalc.rad(angleDeg);

        const xStart = cx + rInner * Math.cos(angleRad);
        const yStart = cy + rInner * Math.sin(angleRad);
        const xEnd = cx + rOuter * Math.cos(angleRad);
        const yEnd = cy + rOuter * Math.sin(angleRad);

        // Ev в”њДџizgisi
        const strokeColor = i === 0 ? "var(--coral)" : "#1f2937";
        const strokeWidth = i === 0 ? "2" : "1";
        svg.innerHTML += `<line x1="${xStart}" y1="${yStart}" x2="${xEnd}" y2="${yEnd}" stroke="${strokeColor}" stroke-width="${strokeWidth}" />`;

        // Burв”њДџ sembollerini dв”Ђв–’в”јЕџ halkaya в”њДџiz
        const midSignLong = cuspLong + 15;
        const midAngleDeg = 180 - (midSignLong - ascLong);
        const midAngleRad = AstroCalc.rad(midAngleDeg);
        const xText = cx + 205 * Math.cos(midAngleRad);
        const yText = cy + 205 * Math.sin(midAngleRad);

        const signInfo = AstroCalc.zodiacs[(ascSignIdx + i) % 12];
        svg.innerHTML += `<text x="${xText}" y="${yText}" text-anchor="middle" dominant-baseline="central" fill="var(--gold)" font-size="14" font-family="'Outfit', sans-serif" style="user-select:none;">${signInfo.glyph}</text>`;

        // Ev Numaralarв”Ђв–’nв”Ђв–’ в”Ђв–‘в”њДџ в”њГ§emberin Hemen Dв”Ђв–’в”јЕџв”Ђв–’na в”њГ§iz
        const xHouseNum = cx + 115 * Math.cos(midAngleRad);
        const yHouseNum = cy + 115 * Math.sin(midAngleRad);
        svg.innerHTML += `<text x="${xHouseNum}" y="${yHouseNum}" text-anchor="middle" dominant-baseline="central" fill="rgba(255,255,255,0.2)" font-size="10" style="user-select:none;">${i+1}</text>`;
      }

      // 3. Gezegen Aв”њДџв”Ђв–’larв”Ђв–’ в”њГ§izgilerini в”њГ§iz (Aspects)
      for (let i = 0; i < planets.length; i++) {
        for (let j = i + 1; j < planets.length; j++) {
          const pA = planets[i];
          const pB = planets[j];

          let diff = Math.abs(pA.long - pB.long);
          if (diff > 180) diff = 360 - diff;

          let aspectColor = "";
          // 0в”¬в–‘ Conjunction (kavuв”јЕџum) - в”њДџizgi в”њДџizilmez veya merkeze yakв”Ђв–’n kв”њв•ќв”њДџв”њв•ќk daire
          if (diff <= 8) aspectColor = "var(--gold)"; 
          // 60в”¬в–‘ Sekstil - yeв”јЕџil/teal
          else if (Math.abs(diff - 60) <= 6) aspectColor = "#06d6a0";
          // 90в”¬в–‘ Kare - kв”Ђв–’rmв”Ђв–’zв”Ђв–’
          else if (Math.abs(diff - 90) <= 8) aspectColor = "#ef476f";
          // 120в”¬в–‘ в”њВЈв”њДџgen - mavi
          else if (Math.abs(diff - 120) <= 8) aspectColor = "#8296c9";
          // 180в”¬в–‘ Karв”јЕџв”Ђв–’t - turuncu/kв”Ђв–’rmв”Ђв–’zв”Ђв–’
          else if (Math.abs(diff - 180) <= 8) aspectColor = "#e76f51";

          if (aspectColor !== "") {
            const angleDegA = 180 - (pA.long - ascLong);
            const angleDegB = 180 - (pB.long - ascLong);

            const xA = cx + rInner * Math.cos(AstroCalc.rad(angleDegA));
            const yA = cy + rInner * Math.sin(AstroCalc.rad(angleDegA));
            const xB = cx + rInner * Math.cos(AstroCalc.rad(angleDegB));
            const yB = cy + rInner * Math.sin(AstroCalc.rad(angleDegB));

            svg.innerHTML += `<line x1="${xA}" y1="${yA}" x2="${xB}" y2="${yB}" stroke="${aspectColor}" stroke-width="1" opacity="0.4" />`;
          }
        }
      }

      // 4. Gezegen Konumlarв”Ђв–’nв”Ђв–’ ve Simgelerini в”њГ§iz
      planets.forEach((p, idx) => {
        const angleDeg = 180 - (p.long - ascLong);
        const angleRad = AstroCalc.rad(angleDeg);

        // в”њГ§akв”Ђв–’в”јЕџmayв”Ђв–’ в”њГ‚nlemek iв”њДџin hafif yarв”Ђв–’в”њДџap deв”ЂЕџiв”јЕџimi
        const offsetR = idx % 2 === 0 ? 150 : 165;

        const xPlan = cx + offsetR * Math.cos(angleRad);
        const yPlan = cy + offsetR * Math.sin(angleRad);

        // Gezegen Simgesi
        svg.innerHTML += `<circle cx="${xPlan}" cy="${yPlan}" r="7" fill="#030610" stroke="#1f2937" stroke-width="1" />`;
        svg.innerHTML += `<text x="${xPlan}" y="${yPlan}" text-anchor="middle" dominant-baseline="central" fill="#fff" font-size="10" font-family="'Outfit', sans-serif" style="user-select:none;">${p.glyph}</text>`;
      });
    }
